7. QUERY :


SELECT SNAME FROM SAILOR S
WHERE NOT EXISTS (
			SELECT B.BID FROM BOAT B
			WHERE NOT EXISTS (
						SELECT R.BID FROM RESERVATION R
						WHERE R.BID = B.BID AND R.SID = S.SID
						)
		      )

OUTPUT :-

SNAME

9. QUERY :-


SELECT S.SID , S.SNAME, B.COLOUR FROM SAILOR S
JOIN RESERVATION R ON S.SID = R.SID
JOIN BOAT B ON B.BID = R.BID AND B.COLOUR = 'RED'
JOIN RESERVATION R1 ON S.SID = R1.SID
JOIN BOAT B1 ON B1.BID = R1.BID AND B1.COLOUR = 'GREEN'

OUTPUT :-

SID	SNAME	COLOUR
210	SAM	RED
