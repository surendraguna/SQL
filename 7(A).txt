DECLARE
	ID INT;
	MID_S NUMBER(10,3);
	SAL NUMBER(10,3);
BEGIN
	ID := &ID;
	SELECT (MAX(SALARY) + MIN(SALARY)) / 2 INTO MID_S FROM EMPLOYEE;
	SELECT SALARY INTO SAL FROM EMPLOYEE WHERE E_ID = ID;
	IF SAL > MID_S THEN 
		UPDATE EMPLOYEE 
		SET SALARY = SAL * 1.08
		WHERE E_ID = ID;
		DBMS_OUTPUT.PUT_LINE('SALARY UPDATED TO ' || SAL * 1.08 || ' FOR E_ID ' || ID);
	ELSE
		UPDATE EMPLOYEE 
		SET SALARY = MID_S
		WHERE E_ID = ID;
		DBMS_OUTPUT.PUT_LINE('SALARY UPDATED TO ' || MID_S || ' FOR E_ID ' || ID);
	END IF;
END;

OUTPUT  :- 

Enter value for id: 505
SALARY UPDATED TO 2358.65 FOR E_ID 505
