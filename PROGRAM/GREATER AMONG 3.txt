/* GREATER NUMBER AMONG THREE */

DECLARE
	A INT;
	B INT;
	C INT;
BEGIN
	A:=&A;
	B:=&B;
	C:=&C;
	IF A > B AND A > C THEN
		DBMS_OUTPUT.PUT_LINE('GREATER NUMBER IS : '||A);
	ELSE IF B > A AND B > C THEN
		DBMS_OUTPUT.PUT_LINE('GREATER NUMBER IS : '||B);
	ELSE 
		DBMS_OUTPUT.PUT_LINE('GREATER NUMBER IS : '||C);
	END IF;
	END IF;
END;

OUTPUT :

ENTER VALUE FOR A : 1
ENTER VALUE FOR B : 2
ENTER VALUE FOR C : 3

GREATER NUMBER IS 3
