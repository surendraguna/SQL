/* EVEN AND ODD NUMBERS UPTO N */

DECLARE
	N INT;
BEGIN
	N := &N;
	DBMS_OUTPUT.PUT_LINE('EVEN NUMBERS :- ');
	FOR I IN 1..N
	LOOP
		IF MOD(I,2) = 0 THEN
			DBMS_OUTPUT.PUT_LINE(I);
		END IF;
	END LOOP;
	DBMS_OUTPUT.PUT_LINE('ODD NUMBERS :- ');
	FOR I IN 1..N
	LOOP
		IF MOD(I,2) != 0 THEN
			DBMS_OUTPUT.PUT_LINE(I);
		END IF;
	END LOOP;
END;

OUTPUT :-
