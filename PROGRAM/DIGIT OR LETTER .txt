/* DIGIT OR LETTER */

DECLARE
	A CHAR(1);
	B NUMBER;
BEGIN
	A := '&A';
	B := ASCII(A);
	IF B >= ASCII('0') AND B <= ASCII('9') THEN
		DBMS_OUTPUT.PUT_LINE(A || ' IS A DIGIT');
	ELSE IF (B >= ASCII('A') AND B <= ASCII('Z')) OR (B >= ASCII('a') AND B <= ASCII('z')) THEN
		DBMS_OUTPUT.PUT_LINE(A || ' IS A LETTER');
	ELSE
		DBMS_OUTPUT.PUT_LINE(A||' IS NEITHER DIGIT NOR LETTER');
	END IF;
	END IF;
END;

OUTPUT :- 

Enter value for a: +
+ IS NEITHER DIGIT NOR LETTER
