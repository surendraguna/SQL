/* PALINDROME */

DECLARE
	P VARCHAR(30);
	Q VARCHAR(30);
BEGIN
	P := '&P';
	FOR I IN REVERSE 1..LENGTH(P)
	LOOP
		Q := Q || SUBSTR(P,I,1);
	END LOOP;
	IF P = Q THEN
		DBMS_OUTPUT.PUT_LINE(P || ' IS A PALINDROME');
	ELSE
		DBMS_OUTPUT.PUT_LINE(P || ' IS NOT A PALINDROME');
	END IF;
END;

OUTPUT :-

Enter value for p: REFER
REFER IS A PALINDROME
