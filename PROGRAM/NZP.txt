/* +VE, -VE, 0 */

DECLARE
	I INT;
BEGIN
	I := &I;
	IF ( I = 0) THEN
		DBMS_OUTPUT.PUT_LINE('GIVEN NUMBER IS 0');
	ELSE IF ( I > 0) THEN 
		DBMS_OUTPUT.PUT_LINE('GIVEN NUMBER IS +VE');
	ELSE 
		DBMS_OUTPUT.PUT_LINE('GIVEN NUMBER IS	-VE');
	END IF;
	END IF;
END;

OUTPUT :

Enter value for i : 99
GIVEN NUMBER IS +VE
