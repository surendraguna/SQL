/* ARMSTRONG NUMBER */

DECLARE
	A INT;
	c INT := 0;
	TEMP INT;
	B INT := 0;
	R INT;
BEGIN
	A := &A;
	TEMP := A;
	WHILE TEMP != 0
	LOOP
		TEMP := TEMP/10;
		C := C+1;
	END LOOP;
	TEMP := A;
	WHILE TEMP != 0
	LOOP
		R := MOD(TEMP, 10);
		B := B + POWER(R, C);
		TEMP := TRUNC(TEMP/10);
	END LOOP;
	IF B = A	THEN
		DBMS_OUTPUT.PUT_LINE(A || ' IS A ARMSTRONG');
	ELSE
		DBMS_OUTPUT.PUT_LINE(A || ' IS NOT A ARMSTRONG');
	END IF;
END;
