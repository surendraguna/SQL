/* LOOP */
DECLARE
	I INT := 0;
BEGIN
	DBMS_OUTPUT.PUT_LINE('SIMPLE LOOP');
	LOOP
		DBMS_OUTPUT.PUT_LINE(I);
		I := I+1;
		IF I > 9 THEN
			EXIT;
		END IF;
	END LOOP;
	DBMS_OUTPUT.PUT_LINE('FOR LOOP');
	FOR I IN 0..9
	LOOP
		DBMS_OUTPUT.PUT_LINE(I);
	END LOOP;
	I := 1;
	DBMS_OUTPUT.PUT_LINE('WHILE LOOP');
	WHILE I < 10
	LOOP 
		DBMS_OUTPUT.PUT_LINE(I);
		I := I+1;
	END LOOP;
END;
