CREATE OR REPLACE FUNCTION INC
(
	E_ID IN EMP.EMP_ID%TYPE
)
RETURN DECIMAL
IS
	SALARY1 EMP.SALARY%TYPE;
	SALARY2 EMP.SALARY%TYPE;
BEGIN
	SELECT SALARY INTO SALARY1 FROM EMP
	WHERE EMP_ID = E_ID;
	IF SALARY1 <= 3000 THEN
		SALARY2 := SALARY1 * (1 + 0.3);
	ELSE IF SALARY1 > 3000 AND SALARY1 <= 6000 THEN
		SALARY2 := SALARY1 * (1 + 0.2);
	ELSE
		SALARY2 := SALARY1 * (1 + 0.1);
	END IF;
	END IF;
	UPDATE EMP
	SET SALARY = SALARY2
	WHERE EMP_ID = E_ID;
	RETURN SALARY2 - SALARY1;
END;

DECLARE
	EID EMP.EMP_ID%TYPE := &EID;
BEGIN
    DBMS_OUTPUT.PUT_LINE('INCREMENT ' || INC(EID));
END;
