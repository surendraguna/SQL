DECLARE 
	S_ID STUDENT.STUDENT_ID%TYPE := &S_ID;
	ID STUDENT.STUDENT_ID%TYPE;
	ADDRESS STUDENT.ADDRESS%TYPE;
BEGIN
	SELECT STUDENT_ID, ADDRESS INTO ID, ADDRESS FROM STUDENT
	WHERE STUDENT_ID = s_ID;
	DBMS_OUTPUT.PUT_LINE(ID || ' ' || ADDRESS);
EXCEPTION
	WHEN NO_DATA_FOUND THEN DBMS_OUTPUT.PUT_LINE('NO STUDENT WITH ID : ' || S_ID);
	WHEN OTHERS	THEN DBMS_OUTPUT.PUT_LINE('ERROR');
END;

OUTPUT :- 

Enter value for s_id: 105
NO STUDENT WITH ID : 105
