DECLARE
	NAME EMP.NAME%TYPE;
	SALARY1 EMP.SALARY%TYPE;
	SALARY2 EMP.SALARY%TYPE := &SALARY2;
	CURSOR C1 IS
		SELECT NAME, SALARY FROM EMP
		WHERE SALARY < SALARY2;
BEGIN
	OPEN C1;
	DBMS_OUTPUT.PUT_LINE('NAME' || CHR(9) || 'SALARY');
	LOOP
		FETCH C1 INTO NAME, SALARY1;
		EXIT WHEN C1%NOTFOUND;
		DBMS_OUTPUT.PUT_LINE( NAME || CHR(9) || SALARY1 );
	END LOOP;
	CLOSE C1;
END;

OUTPUT :-

Enter value for salary2: 30000
NAME    SALARY
SURYA   20000
JAI     25000
VIJAY   29000
