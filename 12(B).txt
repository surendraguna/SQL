CREATE TABLE STD1
(
	ID	INT NOT NULL,
	NAME VARCHAR(30) NOT NULL,
	MARK INT CHECK(MARK > 0 AND MARK <= 30)
)

CREATE OR REPLACE TRIGGER CHECKMARK
AFTER 
INSERT 
ON STD1
FOR EACH ROW
DECLARE
	MARK INT := 12;
BEGIN
	IF :NEW.MARK < MARK THEN
		DBMS_OUTPUT.PUT_LINE('Low marks detected! Notify HOD.');
	END IF;
END;	

INSERT INTO STD1 VALUES(&ID, '&NAME', &MARK)

OUTPUT :-

Enter value for id: 502
Enter value for name: SURENDRA
Enter value for mark: 11
Low marks detected! Notify HOD.

1 row created.
