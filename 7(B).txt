DECLARE
	GRADE CHAR := '&GRADE';
	DESCRIPTION VARCHAR(50);
BEGIN 
	DESCRIPTION := CASE GRADE
			WHEN 'A' THEN 'EXCELLENT'
			WHEN 'B' THEN 'GOOD'
			WHEN 'C' THEN 'AVERAGE'
			WHEN 'D' THEN 'BELOW AVERAGE'
			ELSE 'FAIL'
			END;
	DBMS_OUTPUT.PUT_LINE('GRADE ' || GRADE || ' : ' || DESCRIPTION);
END;

OUTPUT :- 

Enter value for grade: A
GRADE A : EXCELLENT
