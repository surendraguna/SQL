DECLARE
	RANGE EMP.SALARY%TYPE;
	SALARY EMP.SALARY%TYPE;
	ABOVE_SALARY_RANGE EXCEPTION;
	BELOW_SALARY_RANGE EXCEPTION;
BEGIN
	SELECT (MAX(SALARY) + MIN(SALARY))/2 INTO RANGE FROM EMP;
	SELECT SALARY INTO SALARY FROM EMP
	WHERE EMP_ID = &EMP_ID;
    	IF SALARY > RANGE THEN
		RAISE ABOVE_SALARY_RANGE;
	ELSE 
		RAISE BELOW_SALARY_RANGE;
	END IF;
EXCEPTION
	WHEN ABOVE_SALARY_RANGE THEN
		DBMS_OUTPUT.PUT_LINE(SALARY || ' > ' || RANGE || ' ABOVE SALARY RANGE');
	WHEN BELOW_SALARY_RANGE THEN
		DBMS_OUTPUT.PUT_LINE(SALARY || ' < ' || RANGE || ' BELOW SALARY RANGE');
END;

29999 < 40277.5 BELOW SALARY RANGE
