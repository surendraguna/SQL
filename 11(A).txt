DECLARE
	NAME EMP1.NAME%TYPE;
	NO_JOBS EMP1.JOB%TYPE;
	CURSOR C1 IS
	SELECT * FROM EMP1;
BEGIN
	OPEN C1;
	DBMS_OUTPUT.PUT_LINE('NAME' || CHR(9) || 'NO_JOBS');
	LOOP
		FETCH C1 INTO NAME, NO_JOBS;
		DBMS_OUTPUT.PUT_LINE(NAME || CHR(9) || NO_JOBS);
	EXIT WHEN C1 %NOTFOUND;
	END LOOP;
END;


OUTPUT :- 

NAME    NO_JOBS
SURYA   3
JAYA    5
VIJAYA  6
