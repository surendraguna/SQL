CREATE OR REPLACE TRIGGER CHECKATTEND
AFTER 
UPDATE
ON STD
FOR EACH ROW
DECLARE
	ATD INT := 0.85;
BEGIN
	IF :NEW.ATTEND < ATD THEN
		DBMS_OUTPUT.PUT_LINE('Low attendance detected! Notify HOD.');
	END IF;
END;	

CREATE TABLE STD
(
	ID	INT	NOT NULL,
	NAME VARCHAR(30) NOT NULL,
	ATTEND NUMBER(5, 2) NOT NULL
)

INSERT INTO STD VALUES(502, 'SURENDRA', 87);

UPDATE STD
SET ATTEND = 0.84
WHERE ID = 502

OUTPUT :-

Low attendance detected! Notify HOD.

1 row updated.
